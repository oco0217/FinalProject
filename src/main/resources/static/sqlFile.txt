#수정완료
CREATE TABLE customer (
	id VARCHAR(150) NOT NULL,
	pw VARCHAR(160),
	nick_name VARCHAR(30) NOT NULL,
	provider VARCHAR(256),
	provider_id VARCHAR(256),
	register_date DATETIME DEFAULT NOW(),
	role VARCHAR(20) DEFAULT 'role_user',
	m_status tinyint default 0,
	m_reg_date datetime default now(),
	point bigint default 0,
	PRIMARY KEY (id)
);

#수정완료
CREATE TABLE seller (
	id VARCHAR(150) NOT NULL,
	pw VARCHAR(160) NOT NULL,
	shop_name VARCHAR(40) NOT NULL,
    tel VARCHAR(100),
    role VARCHAR(20) DEFAULT 'role_seller',
	PRIMARY KEY (id)
);

#수정완료
CREATE TABLE product (
	id bigint NOT NULL AUTO_INCREMENT,
	name VARCHAR(200) NOT NULL,
	price bigint NOT NULL DEFAULT 0,
	discount_rate int default 0,
	discount_price bigint default 0,
	product_category_detail_id bigint NOT NULL,
	description text,
	total_qty bigint,
	total_sold_qty bigint default 0,
	register_date DATETIME DEFAULT NOW(),
	main_image VARCHAR(256),
	thumb_image VARCHAR(256),
	seller_id VARCHAR(50) NOT NULL,
	is_del VARCHAR(5) DEFAULT "N",
	review_count bigint default 0,
	PRIMARY KEY (id)
);

#수정완료
CREATE TABLE orders (
    id bigint AUTO_INCREMENT,
	merchant_uid VARCHAR(256) NOT NULL,
	imp_uid VARCHAR(256),
	customer_id VARCHAR(150) NOT NULL,
	product_id bigint NOT NULL,
	qty bigint NOT NULL,
	original_price bigint NOT NULL,
	pay_price bigint NOT NULL,
	orders_date DATETIME DEFAULT NOW(),
	status VARCHAR(10) default '주문대기',
	#status가 변경될때마다 변경 (삭제하기 위해)
	status_date DATETIME DEFAULT NOW(),
	is_confirmed TINYINT DEFAULT 0,
	is_review_comment tinyint DEFAULT 0,
	PRIMARY KEY(id)
);

#수정완료
CREATE TABLE address (
    ano bigint NOT NULL AUTO_INCREMENT,
    address VARCHAR(400) NOT NULL,
    customer_id VARCHAR(30) NOT NULL,
    isMain VARCHAR(30) DEFAULT 'N',
    addr_nick_name VARCHAR(30) NOT NULL,
    addr_name VARCHAR(30) NOT NULL,
    contact_num VARCHAR(30) NOT NULL,
    PRIMARY KEY(ano)
);

#수정완료
CREATE TABLE review (
	id bigint NOT NULL AUTO_INCREMENT,
	content VARCHAR(200) NOT NULL,
	star bigint NOT NULL,
	register_date DATETIME DEFAULT NOW(),
	rev_up_count bigint DEFAULT 0,
	customer_id VARCHAR(150) NOT NULL,
	nick_name VARCHAR(30) NOT NULL,
	product_id bigint NOT NULL,
	orders_id bigint NOT NULL,
	PRIMARY KEY (id)
);

#리뷰이미지
CREATE TABLE review_image(
    review_id bigint NOT NULL,
    review_image VARCHAR(256) NOT NULL,
    review_thumb_image VARCHAR(256) NOT NULL
);

#리뷰좋아요
CREATE TABLE review_like(
    review_id bigint NOT NULL,
    customer_id VARCHAR(150) NOT NULL
);

#수정완료
CREATE TABLE member_coupon (
	use_status BOOLEAN default 0,
	use_date DATETIME,
	customer_id VARCHAR(150) NOT NULL,
	coupon_id bigint
);

#수정완료
CREATE TABLE coupon (
	id bigint NOT NULL AUTO_INCREMENT,
	name VARCHAR(100) NOT NULL,
	discount_amount bigint,
	issue_date DATETIME DEFAULT NOW(),
    end_date DATETIME,
	PRIMARY KEY (id)
);

#수정완료
CREATE TABLE basket (
	customer_id VARCHAR(150) NOT NULL,
	qty bigint NOT NULL,
	product_id bigint NOT NULL,
	is_checked TINYINT DEFAULT 1
);

#수정완료
CREATE TABLE product_qna (
	id bigint NOT NULL AUTO_INCREMENT,
	title VARCHAR(100) NOT NULL,
	content VARCHAR(256) NOT NULL,
	register_date DATETIME DEFAULT NOW(),
	qna_status BOOLEAN NOT NULL,
	customer_id VARCHAR(150) NOT NULL,
	PRIMARY KEY (id)
);

#수정완료
CREATE TABLE product_qna_answer (
	id bigint NOT NULL AUTO_INCREMENT,

	seller_id VARCHAR(50) NOT NULL,

#수정완료
CREATE TABLE payment (
	merchant_uid VARCHAR(256) NOT NULL,
	customer_id VARCHAR(150) NOT NULL,
	pay_description VARCHAR(100),
	pay_date DATETIME DEFAULT NOW(),
	original_price bigint NOT NULL,
	pay_price bigint NOT NULL,
	pay_method VARCHAR(100),
	address VARCHAR(400),
	status VARCHAR(10) DEFAULT '결제대기',

SELECT *
FROM product
ORDER BY total_sold_qty DESC
LIMIT 300;
